## Mobile Performance Optimizations

### Virtual Scrolling for Long Lists

```typescript
// Mobile-optimized virtual scrolling
interface VirtualScrollConfig {
  itemHeight: number; // Fixed or estimated
  overscan: number; // Items to render outside viewport
  bufferSize: number; // Items to keep in memory
}

function useMobileVirtualScroll<T>(
  items: T[],
  config: VirtualScrollConfig
) {
  const [scrollTop, setScrollTop] = useState(0);
  const containerRef = useRef<HTMLDivElement>(null);

  // Calculate visible range
  const visibleStart = Math.floor(scrollTop / config.itemHeight);
  const visibleEnd = Math.ceil(
    (scrollTop + window.innerHeight) / config.itemHeight
  );

  // Add overscan for smooth scrolling
  const renderStart = Math.max(0, visibleStart - config.overscan);
  const renderEnd = Math.min(
    items.length,
    visibleEnd + config.overscan
  );

  // Only render visible items
  const visibleItems = items.slice(renderStart, renderEnd);

  return {
    containerRef,
    visibleItems,
    totalHeight: items.length * config.itemHeight,
    offsetTop: renderStart * config.itemHeight,
  };
}
```
