# Phase 1: Core Platform - Mobile Implementation Guide

## Executive Summary

HubbleWave takes a mobile-first, Progressive Web App (PWA) approach, delivering a native-like experience without the overhead of separate native applications. This document outlines the mobile implementation strategy for Phase 1 core features.

### React Component Pattern for Mobile

```tsx
// Responsive hook for mobile detection
function useIsMobile(): boolean {
  const [isMobile, setIsMobile] = useState(false);

  useEffect(() => {
    const mediaQuery = window.matchMedia('(max-width: 767px)');
    setIsMobile(mediaQuery.matches);

    const handler = (e: MediaQueryListEvent) => setIsMobile(e.matches);
    mediaQuery.addEventListener('change', handler);
    return () => mediaQuery.removeEventListener('change', handler);
  }, []);

  return isMobile;
}

// Mobile-optimized list component
interface MobileListProps<T> {
  items: T[];
  renderItem: (item: T) => React.ReactNode;
  onSwipeLeft?: (item: T) => void;
  onSwipeRight?: (item: T) => void;
  onLongPress?: (item: T) => void;
}

function MobileList<T>({
  items,
  renderItem,
  onSwipeLeft,
  onSwipeRight,
  onLongPress
}: MobileListProps<T>) {
  const isMobile = useIsMobile();

  if (!isMobile) {
    // Render standard list for desktop
    return <StandardList items={items} renderItem={renderItem} />;
  }

  return (
    <div className="mobile-list" role="list">
      {items.map((item, index) => (
        <SwipeableListItem
          key={index}
          onSwipeLeft={() => onSwipeLeft?.(item)}
          onSwipeRight={() => onSwipeRight?.(item)}
          onLongPress={() => onLongPress?.(item)}
        >
          {renderItem(item)}
        </SwipeableListItem>
      ))}
    </div>
  );
}
```

### Bundle Optimization

```typescript
// Route-based code splitting
const routes = [
  {
    path: '/login',
    component: lazy(() => import('./pages/Login')),
    preload: true // Critical path
  },
  {
    path: '/collections',
    component: lazy(() => import('./pages/Collections')),
  },
  {
    path: '/records/:collection',
    component: lazy(() => import('./pages/RecordList')),
  },
  {
    path: '/records/:collection/:id',
    component: lazy(() => import('./pages/RecordDetail')),
  },
  {
    path: '/ava',
    component: lazy(() => import('./pages/AVA')),
  }
];

// Preload adjacent routes based on user behavior
function useRoutePreloader() {
  const location = useLocation();

  useEffect(() => {
    // Preload likely next routes
    const preloadMap: Record<string, string[]> = {
      '/collections': ['/records/*'],
      '/records/*': ['/records/*/new', '/records/*/*'],
    };

    const toPreload = preloadMap[location.pathname] ?? [];
    toPreload.forEach(path => {
      const route = routes.find(r => matchPath(path, r.path));
      if (route?.component?.preload) {
        route.component.preload();
      }
    });
  }, [location]);
}
```
